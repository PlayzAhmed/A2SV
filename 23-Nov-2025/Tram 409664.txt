# Problem: Tram - https://codeforces.com/problemset/problem/116/A

using namespace std;
#include <iostream>

int main() {

    long long n;
    int a, b, passengers = 0, capacity = 0;
    
    cin >> n;
    
    for(int i = 0; i < n; i++){
        cin >> a >> b;
        
        passengers -= a;
        passengers += b;
        
        if (capacity < passengers) {capacity = passengers;}
    }
    
    cout << capacity;

    return 0;
}
