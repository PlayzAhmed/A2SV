# Problem: Young Physicist - https://codeforces.com/problemset/problem/69/A

#include <iostream>
using namespace std;

int main() {
    int sumX, sumY, sumZ = 0;
    int n;
    cin >> n;
    for(int i = 0; i < n; i++){
        int x, y, z;
        
        cin >> x >> y >> z;
       sumX += x;
       sumY += y;
       sumZ += z;
    }
    if (sumX == 0 && sumY == 0 && sumZ == 0)
        cout << "YES";
    else 
        cout << "NO";
    return 0;
}
